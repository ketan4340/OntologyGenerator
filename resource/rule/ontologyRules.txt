# 概念階層 (rdfs:subClassOf)
IF () {
  ?Stc a jass:Sentence ;
	jass:subject ?Cls_S ;
	jass:predicate ?Cls_P . 
  ?Cls_S jass:categoremMeans ?Cpt_S .
  ?Cls_P jass:categorem ?Wrd_P .
  ?Wrd_P jass:pos "名詞" ;
	jass:means ?Cpt_P .  
}THEN{
  ?Cpt_S rdfs:subClassOf ?Cpt_P .
}

# 総称 (owl:sameAs)
IF (sameAs) {
  ?Stc rdf:type jass:Sentence .
  ?Stc jass:subject ?Cls_S .
  ?Cls_S jass:categoremMeans ?Cpt_S .
  ?Stc jass:predicate ?Cls_P .
  ?Cls_P jass:categorem ?Wrd_P .
  ?Wrd_P jass:pos "名詞" .
  ?Wrd_P jass:infinitive "総称" .
  ?Cls_Pprev ^rdf:first/rdf:rest/rdf:first ?Cls_P .
  ?Cls_Pprev jass:categoremMeans ?Cpt_Pprev .
}THEN{
  ?Cpt_S owl:sameAs ?Cpt_Pprev .
}

# 数値
IF (numeral value) {
  ?Stc rdf:type jass:Sentence .
  ?Stc jass:subject ?Cls_S .
  ?Cls_S jass:categoremMeans ?Cpt_S .
  ?Stc jass:predicate ?Cls_P .
  ?Cls_P jass:categorem ?Wrd_P .
  ?Wrd_P jass:independentWord ?p1 .
}THEN{
  ?Cpt_S sio:SIO_000216 _:bn .
  _:bn rdf:value ?val .
  _:bn sio:SIO_000221 ?unit .
}
