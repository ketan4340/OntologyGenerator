# 総称 (owl:sameAs)
IF (equivalentClass) {
  ?Stc rdf:type jass:Sentence .
  ?Stc jass:subject/jass:categoremMeans ?Cpt_S .
  ?Stc jass:predicate ?SCls_P .
  ?SCls_P jass:consistsOfCategorem 	?Prs .
  ?Prs jass:consistsOfHead/jass:infinitive "総称" .
  ?Prs jass:consistsOfDependent/rdf:rest* ?node .
  ?node rdf:rest rdf:nil ;
	rdf:first/jass:categoremMeans ?Cpt_p .
}THEN{
  ?Cpt_S owl:equivalentClass ?Cpt_p .
}

# 数値
IF (numeral value) {
  ?Stc rdf:type jass:Sentence .
  ?Stc jass:subject ?Cls_S .
  ?Cls_S jass:categoremMeans ?Cpt_S .
  ?Stc jass:predicate ?Cls_P .
  ?Cls_P jass:consistsOfCategorem ?Prs .
#  ?Prs jass:consistsOfDependent/rdf:rest/rdf:first/jass:name ?length .
#  FILTER regex(?length, "長$")
  ?Prs jass:consistsOfHead ?head .
  ?head jass:consistsOfMorphemes/rdf:rest*/rdf:first ?valrsc .
  ?valrsc jass:name ?val .
  FILTER regex(?val, "[0-9０-９]+?")
  ?valrsc ^rdf:first/rdf:rest/rdf:first/jass:name ?unit .
}THEN{
  ?Cpt_S sio:SIO_000216 _:bn .
#  _:bn rdf:type ?length .
  _:bn rdf:value ?val .
  _:bn sio:SIO_000221 ?unit .
}

# 動詞 (jass:Action)
IF (Action-O) {
  ?Stc rdf:type jass:Sentence .
  ?Stc jass:subject ?Cls_S .
  ?Stc jass:predicate ?Cls_P .
  ?Cls_S jass:categoremMeans ?Cpt_S .
  ?Cls_P jass:consistsOfCategorem ?Wrd_P .
  ?Wrd_P jass:mainPoS "動詞" .
  ?Wrd_P jass:means ?Cpt_vb .
}THEN{
  ?Cpt_vb rdf:type jass:Action .
  ?Cpt_vb jass:domain ?Cpt_S .
}
IF (Action-O(サ変)) {
  ?Stc rdf:type jass:Sentence .
  ?Stc jass:subject ?Cls_S .
  ?Stc jass:predicate ?Cls_P .
  ?Cls_S jass:categoremMeans ?Cpt_S .
  ?Cls_P jass:consistsOfCategorem ?Wrd_P .
  ?Wrd_P jass:subPoS1 "サ変接続" .
  ?Wrd_P jass:means ?Cpt_vb .
}THEN{
  ?Cpt_vb rdf:type jass:Action .
  ?Cpt_vb jass:domain ?Cpt_S .
}
IF (Action) {
  ?Stc rdf:type jass:Sentence .
  ?Stc jass:subject ?Cls_S .
  ?Stc jass:predicate ?Cls_P .
  ?Stc jass:object ?Cls_O .
  ?Cls_S jass:categoremMeans ?Cpt_S .
  ?Cls_O jass:categoremMeans ?Cpt_O .
  ?Cls_P jass:consistsOfCategorem ?Wrd_P .
  ?Wrd_P jass:mainPoS "動詞" .
  ?Wrd_P jass:means ?Cpt_vb .
}THEN{
  ?Cpt_vb rdf:type jass:Action .
  ?Cpt_vb jass:domain ?Cpt_S .
  ?Cpt_vb jass:range ?Cpt_O .
}
IF(Action(サ変)){

# 形容詞
IF(attributeOf){
  ?Stc rdf:type jass:Sentence .
  ?Stc jass:subject ?Cls_S .
  ?Stc jass:predicate ?Cls_P .
  ?Cls_S jass:categoremMeans ?Cpt_S .
  ?Cls_P jass:consistsOfCategorem ?Wrd_P .
  ?Wrd_P jass:mainPoS "形容詞" .
  ?Wrd_P jass:means ?Cpt_ad .
}THEN{
  ?Cpt_S jass:attributeOf ?Cpt_ad .
}


# 概念階層 (rdfs:subClassOf)
IF (subClassOf) {
  ?Stc a jass:Sentence ;
	jass:subject ?Cls_S ;
	jass:predicate ?Cls_P . 
  ?Cls_S jass:categoremMeans ?Cpt_S .
  ?Cls_P jass:consistsOfCategorem ?Wrd_P .
  ?Wrd_P jass:mainPoS "名詞" ;
	jass:means ?Cpt_P . 
}THEN{
  ?Cpt_S rdfs:subClassOf ?Cpt_P .
}
