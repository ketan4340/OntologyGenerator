# 形容詞

# 「S(主語リソース)はL(場所)に多い」
IF (a lot at) {
  ?stc a jass:Sentence ;
       jass:subject/jass:categoremMeans ?rsrcS ;
       jass:predicate/jass:consistsOfCategorem  
         [jass:mainPoS "形容詞" ; jass:infinitive "多い"] ;
       jass:caseNIclause ?clsNI .
  ?clsNI jass:consistsOfCategorem [jass:means ?rsrcLoc ; 
                         jass:mainPoS "名詞" ; 
                         jass:subPoS1|jass:subPoS2|jass:subPoS3 "一般" ; 
                         jass:subPoS1 ?hijiritsu] .
  FILTER(?hijiritsu != "非自立")
} THEN {
  ?rsrcS mos:exist [a mos:Existence ;
                    schema:location ?rsrcLoc ;
                    mos:howMany mos:positive] .
};

# 「SはXが多い」
IF (a lot at) {
  ?stc a jass:Sentence ;
       jass:subject/jass:categoremMeans ?rsrcS ;
       jass:predicate/jass:consistsOfCategorem 
         [jass:mainPoS "形容詞" ; jass:infinitive "多い"] ;
       jass:caseGAclause ?clsGA .
  ?clsGA jass:consistsOfCategorem 
           [jass:means ?rsrcO ; jass:mainPoS "名詞"] . 
} THEN {
 _:x rdf:type owl:Class ; 
     owl:intersectionOf (?rsrcS ?rsrcO) ; 
     mos:exist [a mos:Existence ;
                schema:location ?rsrcLoc ;
                mos:howMany mos:positive] .
};

# 主語リソースの一部に形容詞で表す特徴がある
IF (attributeOf+) {
  ?stc a jass:Sentence ;
  	   jass:subject/jass:categoremMeans ?rsrcS ;
  	   jass:predicate ?clsP .
  ?clsP jass:consistsOfCategorem 
        [jass:mainPoS "形容詞" ; jass:means ?rsrcAdj] .
  ?cls_DP jass:dependTo ?clsP ;
          jass:categoremMeans ?rsrcDP ; 
          jass:consistsOfAdjuncts/rdf:rest*/rdf:first
            [jass:mainPoS "助詞" ; jass:infinitive ?prt] .
  FILTER(?prt IN("が","も"))
}THEN{
  ?rsrcS dcterms:hasPart [
      a ?rsrcDP ;
      jass:attributeOf ?rsrcAdj] .
};

# 主語リソースに形容詞で表される特徴がある
IF (attributeOf) {
  ?stc a jass:Sentence ;
       jass:subject/jass:categoremMeans ?rsrcS ;
       jass:predicate ?clsP .
  ?clsP jass:consistsOfCategorem 
          [jass:mainPoS "形容詞" ; jass:means ?rsrcAdj] .
}THEN{
  ?rsrcS jass:attributeOf ?rsrcAdj .
};
