# 形容詞
IF (attributeOf+) {
  ?stc 	a jass:Sentence ;
  		jass:subject ?cls_S ;
  		jass:predicate ?cls_P .
  ?cls_S jass:categoremMeans ?rsc_S .
  ?cls_P jass:consistsOfCategorem 
  			[jass:mainPoS "形容詞" ;	jass:means ?rsc_AD] .
  ?cls_DP 	jass:dependTo ?cls_P ;
  			jass:categoremMeans ?rsc_DP ; 
  			jass:consistsOfAdjuncts/rdf:rest*/rdf:first
				[jass:mainPoS "助詞" ; jass:infinitive ?prt] .
  FILTER(regex(?prt, "[がも]"))
}THEN{
  ?rsc_S jass:hasPart _:bn .
  _:bn a ?rsc_DP .
  _:bn jass:attributeOf ?rsc_AD .
};

IF (attributeOf) {
  ?stc 	a jass:Sentence ;
  		jass:subject ?cls_S ;
  		jass:predicate ?cls_P .
  ?cls_S jass:categoremMeans ?rsc_S .
  ?cls_P jass:consistsOfCategorem 
  			[jass:mainPoS "形容詞" ; jass:means ?rsc_ad] .
}THEN{
  ?rsc_S jass:attributeOf ?rsc_ad .
};
