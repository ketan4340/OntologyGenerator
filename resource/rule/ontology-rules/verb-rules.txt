# 部分 (dcterms:hasPart)
IF (hasPart(をもつ/がある)) {
  ?stc 	a jass:Sentence ;
  		jass:subject/jass:categoremMeans ?rsrcS ;
  		jass:predicate/jass:consistsOfCategorem/jass:infinitive ?has ; 
  		jass:object ?clsO .
  ?clsO 	jass:consistsOfAdjuncts/rdf:first/jass:infinitive ?adjO ;
  			jass:categoremMeans ?rsrcO .
  FILTER( (?has IN("もつ","持つ") && ?adjO="を") || (?has="ある" && ?adjO="が") )
} THEN {
  ?rsrcS dcterms:hasPart ?rsrcO .
};


# 動詞 (jass:Action)
IF (Action) {
	?stc 	a jass:Sentence ; 
  			jass:predicate ?clsP . 
	?clsP 	jass:consistsOfCategorem [jass:mainPoS "動詞" ; jass:means ?rsrcVb] . 
	{
		# 能動態 (受動態ではない)
		OPTIONAL { 
			?clsP 	jass:consistsOfAdjuncts/rdf:first ?1stAdjP . 
			?1stAdjP 	jass:mainPoS "動詞" ; jass:subPoS1 "接尾" ; jass:infinitive ?reruinf .
			FILTER(?reruinf NOT IN("れる","られる")) 
		} 
		?stc jass:subject/jass:categoremMeans ?rsrcS . 
		OPTIONAL { ?stc jass:object/jass:categoremMeans ?rsrcO . }
	} UNION {
		?clsP 	jass:consistsOfAdjuncts/rdf:first ?1stAdjP . 
		?1stAdjP 	jass:mainPoS "動詞" ; jass:subPoS1 "接尾" ; jass:infinitive ?reruinf . 
		# 受動態
		FILTER(?reruinf IN("れる","られる")) 
		?stc jass:subject/jass:categoremMeans ?rsrcO . 
		OPTIONAL{ ?stc jass:object/jass:categoremMeans ?rsrcS . } 
	}
} THEN {
	_:v a ?rsrcVb ; 
	 	schema:agent ?rsrcS ;
  		schema:object ?rsrcO .
	?rsrcVb rdfs:subClassOf schema:Action ;
};
