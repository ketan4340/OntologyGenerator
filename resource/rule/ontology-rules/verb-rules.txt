# 部分 (dcterms:hasPart)
IF (hasPart(をもつ/がある)) {
  ?stc 	a jass:Sentence ;
  		jass:subject/jass:categoremMeans ?rscS ;
  		jass:predicate/jass:consistsOfCategorem/jass:infinitive ?has ; 
  		jass:object ?clsO .
  ?clsO 	jass:consistsOfAdjuncts/rdf:first/jass:infinitive ?adjO ;
  			jass:categoremMeans ?rscO .
  FILTER( (?has IN("もつ","持つ") && ?adjO="を") || (?has="ある" && ?adjO="が") )
} THEN {
  ?rscS dcterms:hasPart ?rscO .
};


# 動詞 (jass:Action)
IF (Action) {
	?stc 	a jass:Sentence ; 
  			jass:predicate ?clsP . 
	?clsP 	jass:consistsOfCategorem [jass:mainPoS "動詞" ; jass:means ?rscVb] . 
	{
		# 能動態
		?stc jass:subject/jass:categoremMeans ?rscS . 
		OPTIONAL{ ?stc jass:object/jass:categoremMeans ?rscO . }
	} UNION {
		# 受動態
		?stc jass:subject/jass:categoremMeans ?rscO . 
		?clsP 	jass:consistsOfAdjuncts/rdf:first 
						[jass:mainPoS "動詞" ; jass:subPoS1 "接尾" ; jass:infinitive ?reruinf] . 
		FILTER(?reruinf IN("れる","られる")) . 
		OPTIONAL{ ?stc jass:object/jass:categoremMeans ?rscS . } 
	}
} THEN {
	_:v a ?srcVb ; 
	 	jass:domain ?rscS ;
  		jass:range ?rscO .
	?rscVb rdfs:subClassOf schema:Action ;
};
